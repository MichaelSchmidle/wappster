# This is a example on how to proxy non-dockerized applications with the file provider.
# The following shows the YAML configuration of the Proxmox Virtual Environment (PVE).
# Users will be redirected to wappster's management port (default: 4443).
# To use this configuration as a template, just copy the file and add the `.yaml` file extension to the copied file.
http:
  routers:
    pve-http:
      entryPoints:
        - http
      middlewares:
        - wapps-mgmt@docker
      rule: Host(`pve.example.com`)
      service: pve
    pve-https:
      entryPoints:
        - https
      middlewares:
        - wapps-mgmt@docker
      rule: Host(`pve.example.com`)
      tls:
        certResolver: wapps
      service: pve
    pve-https-mgmt:
      entryPoints:
        - https-mgmt
      rule: Host(`pve.example.com`)
      tls:
        certResolver: wapps
      service: pve
  services:
    pve:
      loadbalancer:
        servers:
          - url: https://pve1.example.com:8006/
          - ulr: https://pve2.example.com:8006/
